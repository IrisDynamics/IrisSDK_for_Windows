196 void run_in() {
197
198
199 modbus_client.run_in();
200
201 if ( modbus_client.is_response_ready() ) {
202 response = modbus_client.dequeue_transaction();
203 new_data_flag = true; // communicate to other layers that new data was received
204
205 if ( !response->is_reception_valid() ) {
206 cur_consec_failed_msgs++;
207 failed_msg_counter++;
208 if(connection_state == connected && cur_consec_failed_msgs >= connection_config.max_consec_failed_msgs){
209 disconnect();
210 }
211 }
212
213 // Response was valid
214 else {
215
216 cur_consec_failed_msgs = 0;
217 success_msg_counter++;
218
219 switch (response->get_rx_function_code()) {
220
221 case read_holding_registers:{
222 // add the received data to the local copy of the memory map
223 u16 register_start_address = (response->get_tx_data()[0] << 8) + response->get_tx_data()[1];
224 u16 num_registers = (response->get_tx_data()[2] << 8) + response->get_tx_data()[3];
225 for ( int i = 0; i < num_registers; i++ ) {
226 u16 register_data = (response->get_rx_data()[1 + i*2] << 8) + response->get_rx_data()[2 + i*2];
227 orca_reg_contents[register_start_address + i] = register_data;
228 }
229 break;
230 }
231 case write_single_register:
232 // nothing to do
233 break;
234
235 case motor_command:
236 orca_reg_contents[POS_REG_H_OFFSET] = (response->get_rx_data()[ 0] << 8) | response->get_rx_data()[ 1];
237 orca_reg_contents[POS_REG_OFFSET] = (response->get_rx_data()[ 2] << 8) | response->get_rx_data()[ 3];
238 orca_reg_contents[FORCE_REG_H_OFFSET] = (response->get_rx_data()[ 4] << 8) | response->get_rx_data()[ 5];
239 orca_reg_contents[FORCE_REG_OFFSET] = (response->get_rx_data()[ 6] << 8) | response->get_rx_data()[ 7];
240 orca_reg_contents[POWER_REG_OFFSET] = (response->get_rx_data()[ 8] << 8) | response->get_rx_data()[ 9];
241 orca_reg_contents[TEMP_REG_OFFSET] = (response->get_rx_data()[10]);
242 orca_reg_contents[VOLTAGE_REG_OFFSET] = (response->get_rx_data()[11] << 8) | response->get_rx_data()[12];
243 orca_reg_contents[ERROR_REG_OFFSET] = (response->get_rx_data()[13] << 8) | response->get_rx_data()[14];
244 break;
245
246 case read_coils :
247 case read_discrete_inputs :
248 case read_input_registers :
249 case write_single_coil :
250 case read_exception_status :
251 case diagnostics :
252 case get_comm_event_counter :
253 case get_comm_event_log :
254 case write_multiple_coils :
255 case write_multiple_registers :
256 case report_server_id :
257 case mask_write_register :
258 case read_write_multiple_registers:
259 default:
260 // todo: warn about un-implemented function codes being received
261 break;
262 }
263 }

591 void enqueue_motor_frame() {
592
593
594 if(modbus_client.get_queue_size() >= 2) return;
595
596 switch(comms_mode){
597 case SleepMode:
598 motor_command_fn(connection_config.server_address, 0, 0); //any register address other than force or position register_adresses will induce sleep mode and provided register_value will be ignored
599 break;
600 case ForceMode: {
601 if(uint32_t(modbus_client.get_system_cycles() - stream_timeout_start) > stream_timeout_cycles){ //return to sleep mode if stream timed out
602 comms_mode = SleepMode;
603 }
604 else {
605 motor_command_fn(connection_config.server_address, FORCE_CMD, force_command);
606 }
607 break;
608 }
609 case PositionMode:
610 if(!POS_CTRL){
611 return;
612 }
613
614 if(uint32_t(modbus_client.get_system_cycles() - stream_timeout_start) > stream_timeout_cycles){ //return to sleep mode if stream timed out
615 comms_mode = SleepMode;
616 }
617 else {
618 motor_command_fn(connection_config.server_address, POS_CMD, position_command);
619 }
620 break;
621 }

654 int motor_command_fn(uint8_t device_address, uint8_t register_address, int32_t register_value) {
655 uint8_t data_bytes[5] = { uint8_t(register_address),
