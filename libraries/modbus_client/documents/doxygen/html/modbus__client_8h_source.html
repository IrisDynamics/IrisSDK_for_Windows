148 // Looking for message parsing? Check the application run_in() function (e.g. Actuator object).

151 // No timers are enabled
152 if ( my_enabled_timer == TIMER_ID::none ) {
153
154 }
155 // A timer is enabled
156 else {
157
158 Transaction * active_transaction = messages.get_active_transaction();
159
160
161 TIMER_ID expired_timer = has_timer_expired();
162 switch (expired_timer) {
163
164 case TIMER_ID::repsonse_timeout:
165 enable_interframe_delay(); // will allow run_out to send the next message once this expires (note this disables the current timer)
166 increment_diag_counter(return_server_no_response_count);
167 active_transaction->invalidate(Transaction::RESPONSE_TIMEOUT_ERROR);
168 active_transaction->mark_finished();
169 break;
170
171 case TIMER_ID::interchar_timeout:
172
173 enable_interframe_delay(); // will allow run_out to send the next message once this expires (note this disables the current timer)
174
175 // If the length was unknown assume this was the expected termination of the response until it is validated
176 if( !active_transaction->is_expected_length_known() ){
177 validate_response(active_transaction);
178 }
179 // If the length was known and an interchar timeout occurred (ie the message got messed up)
180 else {
181 increment_diag_counter(unexpected_interchar);
182 active_transaction->invalidate(Transaction::INTERCHAR_TIMEOUT_ERROR);
183 increment_diag_counter(ignoring_state_error);
184 my_state = ignoring;
185 }
186
187 active_transaction->mark_finished();
188
189 break;
190
191
192 case TIMER_ID::turnaround_delay:
193 enable_interframe_delay();
194 break;
195
196
197 case TIMER_ID::interframe_delay: // handled by run_out
198 case TIMER_ID::none: // No timers have expired, but one is enabled
199 default:
200 break;
201 }

212 void run_out () {
213
214 // If the interframe has expired, or there are no timers expired, check for a message to transmit
215 if ( my_enabled_timer == TIMER_ID::none || has_timer_expired() == TIMER_ID::interframe_delay) {
216 disable_timer();
217 if ( messages.available_to_send() ) {
218 my_state = emission;
219 enable_response_timeout();
220 tx_enable(); // enabling the transmitter interrupts results in the send() function being called until the active message is fully sent to hardware
221 increment_diag_counter(message_sent_count); //temp? - for frequency benchmarking
222 }
223 else {
224 my_state = idle;
225 }

355
356 //send the current data byte
357 uint8_t data = active_transaction->pop_tx_buffer();
358 send_byte(data);
359 increment_diag_counter(bytes_out_count);
360
361 if ( active_transaction->is_fully_sent() ) {
362
363 if ( active_transaction->is_broadcast_message() ){ //is it a broadcast message?
364 enable_turnaround_delay();
365 }else{
366 enable_response_timeout();
367 }
368
369 tx_disable(); // enable receiver, disable further transmitter interrupts
370 my_state = reception;
371 }

431 void validate_response(Transaction* response){
432
433 // Check that destination and source addresses are the same
434 if(response->get_tx_address() != response->get_rx_address()){ //if response to broadcast or incorrect responder
435 increment_diag_counter(unexpected_responder);
436 response->invalidate(Transaction::UNEXPECTED_RESPONDER); // invalidates message
437 }
438
439 // Checking CRC
440 if ( !response->check_rx_buffer_crc() ) {
441 increment_diag_counter(crc_error_count);
442 response->invalidate(Transaction::CRC_ERROR); // invalidates message
443 }
444
445 // todo; other things that might invalidate a message
446
447
448 // Increment counters for valid messages
449 if ( response->is_reception_valid() ) {
450 increment_diag_counter(return_bus_message_count);
451
452
453 // Parse exception responses
454 if (response->is_error_response()) {
455 increment_diag_counter(return_server_exception_error_count);
456 switch(response->get_rx_data()[0]){
457 case 5: //exception code corresponding to NAK
458 increment_diag_counter(return_server_NAK_count);
459 break;
460 case 6: //exception code corresponding to server busy error
461 increment_diag_counter(return_server_busy_count);
462 }
463 }

528 TIMER_ID has_timer_expired () {
529
530 volatile uint32_t timer_start_time_l = timer_start_time;
531 volatile uint32_t tnow = get_system_cycles();
532
533 switch (my_enabled_timer) {
534 case TIMER_ID::repsonse_timeout : if ((uint32_t)(tnow - timer_start_time_l) > repsonse_timeout_cycles ) return TIMER_ID::repsonse_timeout ; break;
535 case TIMER_ID::interchar_timeout: if ((uint32_t)(tnow - timer_start_time_l) > interchar_timeout_cycles ) return TIMER_ID::interchar_timeout; break;
536 case TIMER_ID::turnaround_delay : if ((uint32_t)(tnow - timer_start_time_l) > turnaround_delay_cycles ) return TIMER_ID::turnaround_delay ; break;
537 case TIMER_ID::interframe_delay : if ((uint32_t)(tnow - timer_start_time_l) > interframe_delay_cycles ) return TIMER_ID::interframe_delay ; break;
538 case TIMER_ID::none:
539 default:
540 break;
541 }
