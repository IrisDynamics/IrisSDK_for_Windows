186 bool min_powers(Actuator* actuator){
187 int targets_reached = 0;
188 int force_avg[3];
189 force_avg[front] = (actuator[2].get_force_mN() + actuator[3].get_force_mN())>>1;
190 force_avg[mid] = (actuator[1].get_force_mN() + actuator[4].get_force_mN())>>1;
191 force_avg[rear] = (actuator[0].get_force_mN() + actuator[5].get_force_mN())>>1;
192 targets_reached += min_power(front, force_avg[front]) + min_power(mid, force_avg[mid]) + min_power(rear, force_avg[rear]);
193 if (targets_reached == 3) return true;
194 else return false;
195 }
196
197 bool min_power(int section, int force_avg){
198 switch(min_power_states[section]){
199 case hold:
200 drain_tank(section, false);
201 fill_tank(section, false);
202 if (force_avg > ( FORCE_ENVELOPE)) min_power_states[section] = up;
203 else if (force_avg < (-FORCE_ENVELOPE)) min_power_states[section] = down;
204 return true;
205 break;
206 case up:
207 if (force_avg <= 0) min_power_states[section] = hold;
208 else fill_tank(section, true);
209 return false;
210 break;
211 case down:
212 if (force_avg >= 0) min_power_states[section] = hold;
213 else drain_tank(section, true);
214 return false;
215 break;
216 }
217 return false;
218 }

221 bool seek_targets(){
222 int targets_reached = 0;

224 if (targets_reached == 3) return true;

228 /*
229 * @brief Seek to reach a target pressure, hysteresis to not bounce around
230 * @param tank section to seek on
231 * @return true if holding target if seeking false
232 */
233 bool seek_target(int section){
234 switch(seeking_states[section]){
235 case hold:
236 drain_tank(section, false);
237 fill_tank(section, false);
238 if (get_tank_pressure_PSI(section) < (target_pressure[section] - PRESSURE_ENVELOPE)) seeking_states[section] = up;
239 else if (get_tank_pressure_PSI(section) > (target_pressure[section] + PRESSURE_ENVELOPE)) seeking_states[section] = down;
240 return true;
241 break;
242 case up:
243 if (get_tank_pressure_PSI(section) >= target_pressure[section]) seeking_states[section] = hold;
244 else fill_tank(section, true);
245 return false;
246 break;
247 case down:
248 if (get_tank_pressure_PSI(section) <= target_pressure[section]) seeking_states[section] = hold;
249 else drain_tank(section, true);
250 return false;
251 break;
252 }
253 return false;

294 void run_in() {
295
296 modbus_client.run_in();
297
298 if ( modbus_client.is_response_ready() ) {
299
300 response = modbus_client.dequeue_transaction();
301 new_data_flag = true; // communicate to other layers that new data was received
302
303 if ( !response->is_reception_valid() ) {
304 cur_consec_failed_msgs++;
305 failed_msg_counter++;
306 if(connection_state == connected && cur_consec_failed_msgs >= connection_config.max_consec_failed_msgs){
307 disconnect();
308 }
309 }
310
311 // Response was valid
312 else {
313
314 cur_consec_failed_msgs = 0;
315 success_msg_counter++;
316
317 switch (response->get_rx_function_code()) {
318
319 case read_holding_registers:{
320 // add the received data to the local copy of the memory map
321 u16 num_registers = response->get_rx_data()[0];
322 u16 register_address = uint16_t(response->get_tx_data()[0] << 8) + response->get_tx_data()[1];
323 int reg_index = 0;
324 for (int i = 1; i<num_registers; i +=2){
325 u16 register_data = uint16_t(response->get_rx_data()[i] << 8) + response->get_rx_data()[i+1];
326 pneumatic_reg_contents[register_address+reg_index] = register_data;
327 reg_index++;
328 }
329
330 break;
331 }
332 case write_single_register:
333 // nothing to do
334 break;
335
336 case read_coils :
337 case read_discrete_inputs :
338 case read_input_registers :
339 case write_single_coil :
340 case read_exception_status :
341 case diagnostics :
342 case get_comm_event_counter :
343 case get_comm_event_log :
344 case write_multiple_coils :
345 case write_multiple_registers :
346 case report_server_id :
347 case mask_write_register :
348 case read_write_multiple_registers:
349 default:
350 // todo: warn about un-implemented function codes being received
351 break;
352 }
353 }
