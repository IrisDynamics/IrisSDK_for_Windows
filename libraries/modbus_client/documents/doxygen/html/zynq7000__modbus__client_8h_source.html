178 * When the transmitter FIFO and THRE interrupt are enabled (FCR0 = 1, IER1 = 1), transmit interrupts occur as
179follows:
1801. The transmitter holding register empty interrupt [IIR (3 −0) = 2] occurs when the transmit FIFO is empty. It is
181cleared [IIR (3 −0) = 1] when the THR is written to (1 to 16 characters may be written to the transmit FIFO
182while servicing this interrupt) or the IIR is read.
1832. The transmitter holding register empty interrupt is delayed one character time minus the last stop bit time

220 void adjust_baud_rate(uint32_t baud_rate_bps) override {
221// XUartNs550_SetBaud(uart_base_address, my_clock_freq, baud_rate_bps); //todo: use this instead?
223
224 u32 Divisor = my_clock_freq / (16 * baud_rate_bps);
225 u32 BaudLSB = Divisor & XUN_DIVISOR_BYTE_MASK;
226 u32 BaudMSB = (Divisor >> 8) & XUN_DIVISOR_BYTE_MASK;
227
228// u64 baud = BaudMSB << 16 | BaudLSB;
229
230
231 // Set DLAB to allow divisor latch access
232 u32 LcrRegister = Xil_In32(uart_base_address + XUN_LCR_OFFSET);
233 Xil_Out32(uart_base_address + XUN_LCR_OFFSET, LcrRegister | XUN_LCR_DLAB );
234 // Set the baud Divisors to set rate, the initial write of 0xFF is
235 // to keep the divisor from being 0 which is not recommended as per
236 // the NS16550D spec sheet
237 Xil_Out32(uart_base_address + XUN_DRLS_OFFSET, 0xFF);
238 Xil_Out32(uart_base_address + XUN_DRLM_OFFSET, BaudMSB);
239 Xil_Out32(uart_base_address + XUN_DRLS_OFFSET, BaudLSB);
240 // Restore line control register to allow normal register access
241 Xil_Out32(uart_base_address + XUN_LCR_OFFSET, LcrRegister );

254 void uart_isr() {
255
256 volatile u32 isr_status = Xil_In32(uart_iir_address) & XUN_INT_ID_MASK;
257
258 // https://www.xilinx.com/support/documentation/ip_documentation/axi_uart16550/v2_0/pg143-axi-uart16550.pdf
259 // table 2-8
260 #define MODEM_STATUS 0b0000 // reset by reading modem status register
261 #define TX_HOLDING_REG_EMPTY 0b0010 // reset by reading IIR register (if source of interrupt), or writing inro transmitter holding register
262 #define CHARACTER_TIMEOUT 0b1100 // reset by reading receiver
263 #define RX_DATA_AVAILABLE 0b0100 // reset by reading data from FIFO
264 #define RX_LINE_STATUS 0b0110 // reset by reading LSR
265
266 switch ( isr_status ) {
267
268 case CHARACTER_TIMEOUT:
269 case RX_DATA_AVAILABLE:
270
271 while( byte_ready_to_receive() ) {
272 receive();
273 }
274 break;
275
276 case TX_HOLDING_REG_EMPTY:
277
278 for(int i=0; i < XUN_FIFO_SIZE; i++){
279 if (my_state == emission) send();
280 else { break; }
281 }
282 break;
283
284 case RX_LINE_STATUS:
285 case MODEM_STATUS:
286 default:
287 increment_diag_counter(unhandled_isr);
288 break;
289 }
290 }
