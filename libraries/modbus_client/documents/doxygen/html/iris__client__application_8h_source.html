215 void modbus_handshake() {
216
217 switch (connection_state) {
218 case disconnected:
219 if (UART.get_queue_size() == 0 && has_pause_timer_expired()) {
220 is_paused = false;
221 new_data(); // clear new data flag
222 num_discovery_pings_received = 0;
223 enqueue_ping_msg();
224
225 connection_state = discovery;
226 } else {
227 // run_in() should be pulling any residual messages off the queue as they are received or timeout.
228 }
229 break;
230
231 case discovery:
232
233 // Note that the run_in() function should claim any responses that are ready,
234 // set a new_data flag and save the transaction to the response member
235 if (new_data()) {
236 if (response->is_echo_response() && response->is_reception_valid()) {
237
238 num_discovery_pings_received++;
239
240 if (num_discovery_pings_received >= connection_config.req_num_discovery_pings) {
241 synchronize_memory_map(); // loads many read messages onto the queue
242 connection_state = synchronization;
243 } else {
244 enqueue_ping_msg();
245 }
246 }
247 // new response was failed, or the wrong kind of response
248 else {
249 disconnect();
250 }
251 }
252 break;
253
254
255 case synchronization:
256
257 if (new_data() && !response->is_reception_valid()) {
258 // this allows queued data to timeout or be received before reattempting a connection
259 disconnect();
260 }
261 else {
262 if (UART.get_queue_size() == 0) {
263 enqueue_change_connection_status_fn(
264 connection_config.server_address,
265 true,
266 connection_config.target_baud_rate_bps,
267 connection_config.target_delay_us);
268 connection_state = negotiation;
269 }
270 }
271 break;
272
273
274 case negotiation:
275
276 if (new_data()) {
277
278 // Server responded to our change connection request with its realized baud and delay
279 if (response->get_rx_function_code() == change_connection_status && response->is_reception_valid()) {
280 uint8_t* rx_data = response->get_rx_data();
281 UART.adjust_baud_rate(
282 (uint32_t(rx_data[2]) << 24)
283 | (uint32_t(rx_data[3]) << 16)
284 | (uint32_t(rx_data[4]) << 8)
285 | (uint32_t(rx_data[5]) << 0)); //set baud
286
287 UART.adjust_interframe_delay_us(
288
289 (uint16_t(rx_data[6]) << 8) | rx_data[7]); //set delay
290
291 // Reduce timeouts
292 UART.adjust_response_timeout(connection_config.response_timeout_us);
293
294 connection_state = connected;
295
296 }
297 // Server failed to respond to our change connection request
298 else {
299 disconnect();
300 }
301 }
302 case connected:
303 // connection successful
304 break;
305 }
306 }

392 int enqueue_change_connection_status_fn(uint8_t device_address, bool connect, uint32_t baud_rate_bps, uint16_t delay_us) {
393 Transaction transaction;
394 uint16_t requested_state;
395 connect ? requested_state = 0xFF00 : requested_state = 0;
396
397 uint8_t data[8] = { uint8_t(requested_state >> 8),
398 uint8_t(requested_state),
399 uint8_t(baud_rate_bps >> 24),
400 uint8_t(baud_rate_bps >> 16),
401 uint8_t(baud_rate_bps >> 8),
402 uint8_t(baud_rate_bps),
403 uint8_t(delay_us >> 8),
404 uint8_t(delay_us) };
405
406 transaction.load_transmission_data(device_address, change_connection_status, data, 8, get_app_reception_length(change_connection_status));
407 int check = UART.enqueue_transaction(transaction);
408 return check;
409 }
